三数之和

算法题-三数之和

给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有满足条件且不重复的三元组。

注意：答案中不可以包含重复的三元组。

示例：

给定数组 nums = [-1, 0, 1, 2, -1, -4]，

满足要求的三元组集合为： [ [-1, 0, 1], [-1, -1, 2] ]

来源：力扣（LeetCode） 链接：https://leetcode-cn.com/problems/3sum 著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。

题解：

func threeSum(nums []int) [][]int {
    result := make([][]int,0)
    nums = quickSort(nums,0,len(nums)-1)
    //以第i个数为左边界 i+1为左指针 n-1为右指针 大了 左移右指针 小了 右移左指针
    for i:=0;i<len(nums)-2;i++ {
        left:=i+1
        right:=len(nums)-1
        for {
            if left>=right {
                break
            }
            if nums[left]+nums[right]+nums[i]==0 {
                element := []int{nums[i],nums[left],nums[right]}
                result = append(result,element)
                //移动左指针 使它下个数满足 1.不超过右指针 2.不重复
                for {
                    left++
                    if left>=right || nums[left]!=nums[left-1]{
                        break
                    }
                }
                for {
                    right--
                    if left>=right || nums[right]!=nums[right+1] {
                        break
                    }
                }
            }else if nums[left]+nums[right]+nums[i]>0 {
                right--
            }else {
                left++
            }
        }
        for {
            if i+1<len(nums)-1 && nums[i]==nums[i+1] {
                i++
            }else {
                break
            }
        }
    }
    return result
}
//-4 -1 0 1 3 4 5 6
func quickSort(nums []int,start int,end int) []int {
    if start>=end {
        return nums
    }
    i:=start
    j:=end
    var b bool //b = false represent the key is in left 
    for {
        if i>=j {
            break
        }
        if nums[i]>nums[j] {
            temp:=nums[i]
            nums[i]=nums[j]
            nums[j]=temp
            b=!b
            if b {
                j--
            }else {
                i++
            }
        }else {
            if b {
                j--
            }else {
                i++
            }
        }
    }
    quickSort(nums,start,i-1)
    quickSort(nums,i+1,end)
    return nums
}
