## 算法题 - 括号生成

###### 给出 n 代表生成括号的对数，请你写出一个函数，使其能够生成所有可能的并且有效的括号组合。

###### 例如，给出 n = 3，生成结果为：

```
[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]
```

我的解法：

```
var len int
var mark []int
var result []string
//类似于dfs 全排列问题
func generateParenthesis(n int) []string {
	len=2*n
	ar := make([]byte,len)
	mark = make([]int,len)
	result = make([]string,0)
	createString(ar,0,0)
	return result
}

func createString(ar []byte,pos int,sum int) {
	if  pos==len {
		if sum==0 {
			result = append(result,string(ar[:]))
		}
		return
	}
	if sum==0 {
		ar[pos]='('
		createString(ar,pos+1,sum+1)
	}else if sum==len/2 {
		ar[pos]=')'
		createString(ar,pos+1,sum-1)
	}else {
		ar[pos]='('
		createString(ar,pos+1,sum+1)
		ar[pos]=')'
		createString(ar,pos+1,sum-1)
	}
	return
}
```
